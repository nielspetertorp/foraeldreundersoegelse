<script>

  const APPS_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbz2NWfSDdarj0YVp3y5S6tsAzlU07Sj7iV7nF6NnN00lNXBfC5Mq6-1c14i8taWHZ9L/exec";

 

  const form = document.getElementById("survey");

  const btn = document.getElementById("submitBtn");

  const status = document.getElementById("status");

 

  // Sæt form action dynamisk

  form.action = APPS_SCRIPT_URL;

 

  // Lyt efter postMessage fra Apps Script (iframe)

  window.addEventListener("message", (event) => {

    const data = event.data || {};

    if (typeof data !== "object" || data.ok === undefined) return;

 

    btn.disabled = false;

 

    if (data.ok) {

      status.innerHTML = `

        <div class="ok">

          Tak! Din besvarelse er sendt.<br><br>

          <button id="againBtn" type="button">Besvar for et barn mere</button>

        </div>

      `;

      form.reset();

      document.getElementById("againBtn").addEventListener("click", () => {

        status.innerHTML = "";

        window.scrollTo({ top: 0, behavior: "smooth" });

      });

    } else {

      status.innerHTML = `<div class="err">Noget gik galt: ${escapeHtml(data.error || "Ukendt fejl")}</div>`;

    }

  });

 

  form.addEventListener("submit", () => {

    status.innerHTML = "";

    btn.disabled = true;

  });

 

  function escapeHtml(str) {

    return String(str).replace(/[&<>"']/g, (m) => ({

      "&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;","'":"&#039;"

    }[m]));

  }

</script>
