<script>

  // Din Apps Script URL

  const APPS_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbz2NWfSDdarj0YVp3y5S6tsAzlU07Sj7iV7nF6NnN00lNXBfC5Mq6-1c14i8taWHZ9L/exec";


  const form = document.getElementById("survey");

  const status = document.getElementById("status");


  const backBtn = document.getElementById("backBtn");

  const nextBtn = document.getElementById("nextBtn");

  const submitBtn = document.getElementById("submitBtn");


  const progressBar = document.getElementById("progressBar");

  const stepTitle = document.getElementById("stepTitle");

  const stepsMeta = document.getElementById("stepsMeta");


  const steps = Array.from(document.querySelectorAll(".step"));

  const totalSteps = steps.length;

  let currentStep = 1;


  form.action = APPS_SCRIPT_URL;


  // Vis status fra redirect (hvis du har Apps Script redirect tilbage til ?status=ok/err)

  (function showStatusFromUrl() {

    const url = new URL(window.location.href);

    const st = url.searchParams.get("status");

    if (!st) return;


    if (st === "ok") {

      status.innerHTML = `<div class="ok">Tak! Din besvarelse er sendt. Du kan nu lukke siden — eller udfylde for et barn mere.</div>`;

      form.reset();

    } else {

      const msg = url.searchParams.get("msg") || "Ukendt fejl";

      status.innerHTML = `<div class="err">Noget gik galt: ${escapeHtml(msg)}</div>`;

    }


    window.history.replaceState({}, "", url.origin + url.pathname);

  })();


  function render() {

    steps.forEach(s => s.classList.add("hidden"));

    const active = steps.find(s => Number(s.dataset.step) === currentStep);

    if (active) active.classList.remove("hidden");


    const pct = ((currentStep - 1) / (totalSteps - 1)) * 100;

    progressBar.style.width = pct + "%";


    stepTitle.textContent = `Trin ${currentStep} af ${totalSteps}`;


    Array.from(stepsMeta.querySelectorAll("span")).forEach(sp => {

      sp.classList.toggle("active", Number(sp.dataset.step) === currentStep);

    });


    backBtn.disabled = currentStep === 1;


    // sidste step: vis submit i stedet for next

    const isLast = currentStep === totalSteps;

    nextBtn.classList.toggle("hidden", isLast);

    submitBtn.classList.toggle("hidden", !isLast);


    // Scroll top for overskuelighed

    window.scrollTo({ top: 0, behavior: "smooth" });

  }


  function validateCurrentStep() {

    const active = steps.find(s => Number(s.dataset.step) === currentStep);

    if (!active) return true;


    // checkValidity virker pr. element (inkl. required radio/select/textarea)

    const inputs = Array.from(active.querySelectorAll("input, select, textarea"));

    for (const el of inputs) {

      // skip disabled/hidden inputs (honeypot ligger udenfor step)

      if (el.disabled) continue;

      if (!el.checkValidity()) {

        el.reportValidity();

        // scroll til feltet

        el.scrollIntoView({ behavior: "smooth", block: "center" });

        return false;

      }

    }

    return true;

  }


  nextBtn.addEventListener("click", () => {

    status.innerHTML = "";

    if (!validateCurrentStep()) return;

    if (currentStep < totalSteps) currentStep += 1;

    render();

  });


  backBtn.addEventListener("click", () => {

    status.innerHTML = "";

    if (currentStep > 1) currentStep -= 1;

    render();

  });


  form.addEventListener("submit", (e) => {

    // Valider sidste step (og i praksis hele formen)

    status.innerHTML = "";

    const ok = validateCurrentStep();

    if (!ok) {

      e.preventDefault();

      return;

    }

    // UX: disable knap så man ikke spammer

    submitBtn.disabled = true;

    submitBtn.textContent = "Sender…";

  });


  // Start rendering

  render();


  function escapeHtml(str) {

    return String(str).replace(/[&<>"']/g, (m) => ({

      "&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;","'":"&#039;"

    }[m]));

  }

</script>
